async function enviarDatos() { if (!nombreValor || !trabajoValor) { alert("Ambos campos son obligatorios") return } try { console.log("Voy a enviar:", { name: nombreValor, job: trabajoValor }) const respuesta = await fetch("https://reqres.in/api/users", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ name: nombreValor, job: trabajoValor }) }) if (!respuesta.ok) { throw new Error(`Error del servidor: ${respuesta.status}`) } const respuestaJson = await respuesta.json() console.log("Respuesta del servidor:", respuestaJson) const p = document.createElement("p") p.textContent = `nombre: ${nombreValor}, trabajo: ${trabajoValor}` form.appendChild(p) inputNombre.value = "" inputTrabajo.value = "" } catch (error) { console.error("Ocurrió un error:", error) alert("Algo falló al enviar los datos. Revisá la consola.") } }
